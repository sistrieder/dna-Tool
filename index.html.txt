const selectedSymptoms = new Set();
const selectedGoals = new Set();

document.querySelectorAll('#symptom-chips .chip').forEach(c => {
  c.addEventListener('click', () => toggleChip(c, selectedSymptoms));
});
document.querySelectorAll('#goal-chips .chip').forEach(c => {
  c.addEventListener('click', () => toggleChip(c, selectedGoals));
});

function toggleChip(el, set) {
  const v = el.dataset.val;
  if (set.has(v)) { set.delete(v); el.classList.remove('selected'); }
  else { set.add(v); el.classList.add('selected'); }
}

function addCustomChip(type) {
  const inputId = type === 'symptom' ? 'custom-symptom' : 'custom-goal';
  const gridId  = type === 'symptom' ? 'symptom-chips' : 'goal-chips';
  const set     = type === 'symptom' ? selectedSymptoms : selectedGoals;
  const input   = document.getElementById(inputId);
  const val     = input.value.trim();
  if (!val) return;
  const chip = document.createElement('span');
  chip.className = 'chip selected';
  chip.dataset.val = val;
  chip.textContent = val;
  set.add(val);
  chip.addEventListener('click', () => toggleChip(chip, set));
  document.getElementById(gridId).appendChild(chip);
  input.value = '';
}

function scoreTest(test, symptoms, goals) {
  let score = 0;
  symptoms.forEach(s => {
    if (test.symptoms.some(ts => s.toLowerCase().includes(ts.toLowerCase()) || ts.toLowerCase().includes(s.toLowerCase()))) score += 2;
  });
  goals.forEach(g => {
    if (test.goals.some(tg => g.toLowerCase().includes(tg.toLowerCase()) || tg.toLowerCase().includes(g.toLowerCase()))) score += 3;
  });
  return score;
}

function getExplanation(test, matchedSymptoms) {
  for (const s of matchedSymptoms) {
    for (const key of Object.keys(test.explanations)) {
      if (key !== 'default' && s.toLowerCase().includes(key.toLowerCase())) {
        return test.explanations[key];
      }
    }
  }
  return test.explanations['default'];
}

function buildSummary(sympArr, goalArr, primary) {
  if (primary.length === 0) return '';
  const topTest = primary[0].name;
  const hasMind = primary.some(t => t.id === 'dna-mind');
  const hasGut  = primary.some(t => t.id === 'dna-gut');
  const hasDiet = primary.some(t => t.id === 'dna-diet');

  let text = `Basierend auf Ihren Angaben`;
  if (sympArr.length > 0) text += ` — ${sympArr.slice(0,2).join(', ')}${sympArr.length > 2 ? ' u.a.' : ''}`;
  text += ` — empfiehlt sich vor allem der <strong>${topTest}</strong>`;
  if (primary.length > 1) text += ` sowie ${primary.slice(1).map(t => `<strong>${t.name}</strong>`).join(' und ')}`;
  text += `. Als zertifizierte Nordic Practitioner bespreche ich die Ergebnisse persönlich mit Ihnen und entwickle einen individuellen Leitfaden.`;
  return text;
}

function generateProfile() {
  const sympArr = Array.from(selectedSymptoms);
  const goalArr = Array.from(selectedGoals);

  if (sympArr.length === 0 && goalArr.length === 0) {
    document.getElementById('result-area').innerHTML = '<p class="error-msg">Bitte mindestens ein Symptom oder ein Ziel auswählen.</p>';
    return;
  }

  const scored = testDb
    .map(t => ({ ...t, score: scoreTest(t, sympArr, goalArr) }))
    .sort((a, b) => b.score - a.score);

  const primary   = scored.filter(t => t.score >= 3);
  const secondary = scored.filter(t => t.score > 0 && t.score < 3);

  let html = '';

  const summary = buildSummary(sympArr, goalArr, primary);
  if (summary) html += `<div class="profile-summary">${summary}</div>`;

  if (primary.length > 0) {
    html += `<p class="results-heading">Empfohlene Tests für Ihr Profil</p>`;
    primary.forEach((t, i) => {
      const matchedS = sympArr.filter(s => t.symptoms.some(ts => s.toLowerCase().includes(ts.toLowerCase()) || ts.toLowerCase().includes(s.toLowerCase())));
      const matchedG = goalArr.filter(g => t.goals.some(tg => g.toLowerCase().includes(tg.toLowerCase()) || tg.toLowerCase().includes(g.toLowerCase())));
      const expl = getExplanation(t, matchedS);
      html += `<div class="result-card ${i === 0 ? 'primary' : ''}">
        <div class="card-top">
          <div>
            <p class="test-name">${t.name}</p>
            <p class="test-tagline">${t.tagline}</p>
          </div>
          ${i === 0 ? '<span class="badge badge-primary">Beste Übereinstimmung</span>' : '<span class="badge badge-secondary">Empfohlen</span>'}
        </div>
        <p class="test-desc">${expl}</p>
        <p class="test-price">${t.price}</p>
        <div class="match-tags">
          ${matchedS.map(s => `<span class="match-tag">↳ ${s}</span>`).join('')}
          ${matchedG.map(g => `<span class="match-tag">◎ ${g}</span>`).join('')}
        </div>
      </div>`;
    });
  }

  if (secondary.length > 0) {
